Question-3:

Write a C program to determine whether a given number is prime or not.

💻 Input Code:

#include <stdio.h>
int main() {
    int prime=1;
    int num;
    
    printf("Enter a number: ");
    scanf("%d",&num);
    
    for(int i=2; i<=num; i++){
        if(num%2==0){
            prime=0;
        }
        break;
    }
    if(prime){
        printf("%d is a prime number.",num);
    }
    else{
        printf("%d is not a prime number.",num);
    }
     return 0;
 }

🖥️ Output Code:

Enter a number: 3
3 is a prime number.
